Building wxWidgets using MSys2 packaging.txt

See https://wiki.archlinux.org/index.php/PKGBUILD
    https://www.archlinux.org/pacman/PKGBUILD.5.html

# Open MSys2 command prompt


# One time build tools installation
pacman -S base-devel
# for building mingw32 packages
pacman -S mingw-w64-i686-toolchain
# for building mingw64 packages
pacman -S mingw-w64-x86_64-toolchain


cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-gtk2
pacman -U mingw-w64-i686-gtk2-2.24.31-2-any.pkg.tar.xz

# Change to MinGW Package git repo.
cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-wxwidgets

# Change to wxWidgets folder
# wxwidgets3.1-git
cd ../mingw-w64-wxmsw3.1-git        # Package was built; and it installs
cd ../mingw-w64-wxgtk3.1-gtk2-git   # Package was built; and it installs  
cd ../mingw-w64-wxgtk3.1-gtk3-git   # Package was built; and it installs 

# wxwidgets3.1
cd ../mingw-w64-wxmsw3.1            # Package was built; and it installs
cd ../mingw-w64-wxgtk3.1-gtk3       # Package was built; and it installs

# wxwidgets3.0-git
cd ../mingw-w64-wxmsw-git           # Package was built; and it installs
cd ../mingw-w64-wxgtk-gtk3-git      # Package was built; and it installs

# wxwidgets3.0
cd ../mingw-w64-wxwidgets           # Package was built; and it installs
cd ../mingw-w64-wxgtk-gtk3          # Package was built; and it installs

# 2.8
cd ../mingw-w64-wxwidgets2.8-git    # Package was built; and it installs
cd ../mingw-w64-wxwidgets2.8        # Package was built; and it installs

# Build the 32 bit and 64 packages
# makepkg-mingw -sLf

# Build only the 32 bit package
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf

# Build only the 64 bit package
#  MINGW_INSTALLS=mingw64 makepkg-mingw -sLf


# Look for the packages
ls mingw-w64-*-*-*-*.pkg.tar.xz

pacman -Qs wxwidgets

# Install a 'local' 32 bit package
pacman -U mingw-w64-i686-wxWidgets2.8-2.8.12-1-any.pkg.tar.xz

# Install a 'local' 64 bit package
pacman -U mingw-w64-x86_64-wxWidgets2.8-2.8.12-1-any.pkg.tar.xz
