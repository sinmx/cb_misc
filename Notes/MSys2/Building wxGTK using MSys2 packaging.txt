Building wxWidgets using MSys2 packaging.txt

See https://wiki.archlinux.org/index.php/PKGBUILD
    https://www.archlinux.org/pacman/PKGBUILD.5.html

# Open MSys2 command prompt


# One time build tools installation
pacman -S base-devel
# for building mingw32 packages
pacman -S mingw-w64-i686-toolchain
# for building mingw64 packages
pacman -S mingw-w64-x86_64-toolchain


# Change to package(s) folder
cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-wxwidgets3.1-git
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf                               # Package was built and installs
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p build_wxgtk3.1-gtk3-git.sh # 
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p build_wxgtk3.1-gtk2-git.sh # Package was built and installs

# Change to package(s) folder
cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-wxwidgets3.1
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf                               # Package was built and installs
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p build_wxgtk3.1-gtk3.sh     # Building
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p build_wxgtk3.1-gtk2.sh     # Package was built and installs


# Change to package(s) folder
cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-wxwidgets-git
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf                               # Package was built and installs
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p build_wxgtk3.0-gtk3-git.sh # Building
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p wxgtk3.0-gtk2-git.pkgbuild #

# Change to package(s) folder
cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-wxwidgets
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf                               # Package was built and installs
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p wxgtk3.0-gtk3.pkgbuild     # Package was built and installs
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p wxgtk3.0-gtk2.pkgbuild     # Package was built and installs


# Change to package(s) folder
cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-wxwidgets2.8
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf                               # Building
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p wxgtk2.8-gtk2.pkgbuild     # 

# Change to package(s) folder
cd C:/Devel/open_source_code/version_control/MINGW_packages-git/mingw-w64-wxwidgets2.8-git
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf                               # 
MINGW_INSTALLS=mingw32 makepkg-mingw -sLf -p wxgtk2.8-gtk2.pkgbuild     #
